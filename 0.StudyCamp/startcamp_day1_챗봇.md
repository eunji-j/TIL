#### 로또번호 생성

```python
import random

-1-
lotto = [] 
numbers = list(range(1, 46)) #리스트생성

a = random.sample(numbers, 6) #비복원추출(숫자6개)
print(sorted(a))

-2-
lotto = []
number = random.randrange(1,46)

for i in range(6):
	while number in lotto:  #비복원추출(숫자6개)
		number = random.randrange(1, 46)
	lotto.append(number)
print(sorted(a))
```
