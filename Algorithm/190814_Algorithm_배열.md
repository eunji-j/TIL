di = [0, 1, 0, -1]
dj = [1, 0, -1, 0]

for k in range(4):

    nj = j + dj[k]
    if ni >= 0 and ni < N and nj >= 0 and nj < N:

- 이진 검색

```python
for t in range(int(input())):
    p, a = map(int, input().split())
    cnt = 0
    
    left, right = 1, p
    while 1:
        x = int((left+right)/2)
        cnt += 1
        if x == a:
            break
        if x < a:
            left = x
        else:
            right = x
   
    print(f'#{t+1} {cnt}')
```
